00285 void HistogramClassDetector< HistogramType >::compute( const Configuration& cfg )
00286 {
00287     const std::size_t data_range = static_cast< std::size_t >( cfg.last - cfg.first + 1 );
00288 
00289     CARNA_ASSERT( data_range > cfg.radius );
00290 
00291     const double sigma = cfg.radius / 2.;
00292 
00293     reset();
00294 
00295  // compute logarithmic data
00296 
00297     double logarithmic_data_peak = std::numeric_limits< double >::min();
00298     std::vector< double > logarithmic_data( data_range );
00299     for( unsigned int sample_index = 0; sample_index < data_range; ++sample_index )
00300     {
00301         logarithmic_data[ sample_index ] = std::log10( static_cast< double >( histogram.counter[ cfg.first + sample_index ] + 1u ) );
00302         logarithmic_data_peak = std::max( logarithmic_data_peak, logarithmic_data[ sample_index ] );
00303     }
00304 
00305  // define Gaussian kernel
00306 
00307     const auto gaussian_coefficient = [&]( unsigned int distance )->double
00308     {
00309         return std::exp( -Math::sq( static_cast< signed int >( distance ) ) / ( 2 * Math::sq( sigma ) ) );
00310     };
00311 
00312  // pre-compute Gaussian coefficients
00313 
00314     std::map< unsigned int, double > gaussian_coefficients;
00315     double gaussian_coefficients_sum = 0.;
00316     for( unsigned int sample_distance = 0u; sample_distance <= cfg.radius; ++sample_distance )
00317     {
00318         gaussian_coefficients[ sample_distance ] = gaussian_coefficient( sample_distance );
00319         gaussian_coefficients_sum += gaussian_coefficients[ sample_distance ];
00320     }
00321     for( auto gaussian_coefficient_itr = gaussian_coefficients.begin(); gaussian_coefficient_itr != gaussian_coefficients.end(); ++gaussian_coefficient_itr )
00322     {
00323         gaussian_coefficient_itr->second /= ( 2 * gaussian_coefficients_sum );
00324     }
00325 
00326  // apply Gaussian kernel
00327 
00328     std::vector< double > smoothed_data( data_range );
00329     for( unsigned int dst_sample_index = 0u; dst_sample_index < data_range; ++dst_sample_index )
00330     {
00331         const auto sample_at_distance = [&]( signed int distance )->double
00332         {
00333             const unsigned int src_sample_index =
00334                 static_cast< unsigned int >( Math::clamp< int >( static_cast< signed int >( dst_sample_index ) + distance, 0, data_range - 1 ) );

00336         };

00338         double result = 0.;
00339 
00340         for( signed int sample_distance = -static_cast< signed int >( cfg.radius ); sample_distance <= static_cast< signed int >( cfg.radius ); ++sample_distance )
00341         {
00342             const double sample = sample_at_distance( sample_distance );
00343             const double gaussian_coefficient = gaussian_coefficients[ std::abs( sample_distance ) ];
00344             const double summand = sample * gaussian_coefficient;
00345             result += summand;
00346         }
00347 
00348         smoothed_data[ dst_sample_index ] = result;
00349     }
00350 
00351  // find local negative peaks through derivative evaluation
00352  // the first and the last samples are assumed to be such peaks explicitly
00353 
00354     std::vector< unsigned int > peak_sample_indices;
00355     peak_sample_indices.push_back( 0u );
00356     double last_sample = smoothed_data[ peak_sample_indices.front() ];
00357     double last_derivative_sample = last_sample;
00358 
00359     for( unsigned int sample_index = 1u; sample_index < data_range; ++sample_index )
00360     {
00361         const double derivative_sample = smoothed_data[ sample_index ] - last_sample;
00362         if( last_derivative_sample < 0 && derivative_sample > 0 )
00363         {
00364             peak_sample_indices.push_back( sample_index );
00365         }
00366         last_sample = smoothed_data[ sample_index ];
00367         last_derivative_sample = derivative_sample;
00368     }
00369 
00370     if( peak_sample_indices.back() != smoothed_data.size() - 1 )
00371     {
00372         peak_sample_indices.push_back( smoothed_data.size() - 1 );
00373     }
00374

00377     classes.reset( new Result() );

00379     {
00380         if( classes->empty() && first_index >= 1u )
00381         {
00382             const Class new_class( 0u, first_index - 1, false );
00383             classes->push_back( new_class );
00384         }
00385         else
00386         if( !classes->empty() && static_cast< unsigned int >( classes->back().last + 1 ) < first_index )
00387         {
00388             const Class new_class( classes->back().last + 1, first_index - 1, false );
00389             classes->push_back( new_class );
00390         }
00391         const Class new_class( first_index, last_index, typical );
00392         classes->push_back( new_class );
00393     };
00394 
00395     auto peak_sample_index_itr = peak_sample_indices.begin();
00396     unsigned int first_class_sample_index = *peak_sample_index_itr;
00397     for( ++peak_sample_index_itr; peak_sample_index_itr != peak_sample_indices.end(); ++peak_sample_index_itr )
00398     {
00399         const unsigned int last_class_sample_index = *peak_sample_index_itr;
00400         unsigned int class_candidate_width = last_class_sample_index - first_class_sample_index + 1u;
00401 
00402         if( class_candidate_width > cfg.minimumTypicalClassWidth )
00403         {
00404             double min_sample = +std::numeric_limits< double >::infinity();
00405             double max_sample = -std::numeric_limits< double >::infinity();
00406             for( unsigned int sample_index = first_class_sample_index; sample_index <= last_class_sample_index; ++sample_index )
00407             {
00408                 max_sample = std::max( max_sample, smoothed_data[ sample_index ] );
00409                 min_sample = std::min( min_sample, smoothed_data[ sample_index ] );
00410             }
00411 
00412             const double logarithmic_height_difference = ( max_sample - min_sample ) / logarithmic_data_peak;
00413             if( logarithmic_height_difference > cfg.minimumLogarithmicHeightDifference )
00414             {
00415                 add_class( first_class_sample_index, last_class_sample_index, true );
00416             }
00417         }
00418 
00419         first_class_sample_index = last_class_sample_index + 1u;
00420     }
00421 
00422     if( classes->empty() )
00423     {
00424         add_class( peak_sample_indices.front(), peak_sample_indices.back(), false );
00425     }
00426     else
00427     if( static_cast< unsigned int >( classes->back().last ) < peak_sample_indices.back() )
00428     {
00429         add_class( classes->back().last + 1u, peak_sample_indices.back(), false );
00430     }
00431 
00432  // add offset to classes
00433 
00434     for( auto class_itr = classes->begin(); class_itr != classes->end(); ++class_itr )
00435     {
00436         class_itr->first += cfg.first;
00437         class_itr->last += cfg.first;
00438     }
