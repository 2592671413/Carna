00327 {
00328     counter.reset();

00330     const Vector& end_position = fieldSize;
00331     Vector position = Vector::Zero();
00332 
00333     bool position_moved;
00334     do
00335     {
00336         position_moved = false;
00337 
00338      // update the counter for the current location
00339 
00340         const FieldCodomain value = field( position );
00341         ++counter[ value ];
00342 
00343      // iterate over all vector components and increment the first that hasn't reached it's maximum yet
00344 
00345         for( unsigned int incremented_dimension = 0; incremented_dimension < SCALAR_FIELD_DIMENSION; ++incremented_dimension )
00346         {
00347             if( position[ incremented_dimension ] + 1 < end_position[ incremented_dimension ] )
00348             {
00349                 ++position[ incremented_dimension ];
00350

00352              // e.g. the first and second vector components will be reset whenever the third vector component increases

00355                 {
00356                     if( lower_dimension != incremented_dimension )

00360                 }
