305 template< typename SegmentHUVolumeType, typename SegmentNormalsVolumeType >
306 base::math::Vector3ui VolumeGridHelper< SegmentHUVolumeType, SegmentNormalsVolumeType >::computeMaxSegmentSize
307  ( const base::math::Vector3ui& resolution, std::size_t maxSegmentBytesize )
308 {
309  const float maxSideLengthF = std::pow
310  ( maxSegmentBytesize / static_cast< float >( sizeof( typename SegmentHUVolumeType::Voxel ) ), 1.f / 3 );
311  const unsigned int maxSideLength = base::math::makeEven( base::math::round_ui( maxSideLengthF ), -1 );
312 
313  /* We subtract the redundant texels from effective segment size.
314  * Note that this causes the effective maximum segment size to be odd.
315  */
316  return base::math::Vector3ui( maxSideLength - 1, maxSideLength - 1, maxSideLength - 1 );
317 }

320 template< typename SegmentHUVolumeType, typename SegmentNormalsVolumeType >
321 VolumeGridHelper< SegmentHUVolumeType, SegmentNormalsVolumeType >::VolumeGridHelper
322  ( const base::math::Vector3ui& nativeResolution
323  , std::size_t maxSegmentBytesize )
324  : nativeResolution( nativeResolution )
325  , resolution( base::math::makeEven( nativeResolution, +1 ) )
326  , maxSegmentBytesize( maxSegmentBytesize )
327  , maxSegmentSize( computeMaxSegmentSize( resolution, maxSegmentBytesize ) )
328  , regularSegmentSize( maxSegmentSize.cwiseMin( resolution ) )
329 {
330  const base::math::Vector3ui tails
331  ( resolution.x() % maxSegmentSize.x()
332  , resolution.y() % maxSegmentSize.y()
333  , resolution.z() % maxSegmentSize.z() );
334 
335  const base::math::Vector3ui segmentCounts
336  ( resolution.x() / maxSegmentSize.x() + ( tails.x() > 0 ? 1 : 0 )
337  , resolution.y() / maxSegmentSize.y() + ( tails.y() > 0 ? 1 : 0 )
338  , resolution.z() / maxSegmentSize.z() + ( tails.z() > 0 ? 1 : 0 ) );
339  myGrid.reset( new base::VolumeGrid< SegmentHUVolumeType, SegmentNormalsVolumeType >( maxSegmentSize, segmentCounts ) );
340  NormalsComponent::setGrid( *myGrid );
341 
342  CARNA_FOR_VECTOR3UI( segmentCoord, myGrid->segmentCounts )
343  {
344  /* Here we add the redundant texels to the buffer size considerations. This
345  * is fine because 'regularSegmentSize' contains the effective segment size.
346  */
347  const base::math::Vector3ui segmentSize
348  ( segmentCoord.x() + 1 == myGrid->segmentCounts.x() ? tails.x() : regularSegmentSize.x() + 1
349  , segmentCoord.y() + 1 == myGrid->segmentCounts.y() ? tails.y() : regularSegmentSize.y() + 1
350  , segmentCoord.z() + 1 == myGrid->segmentCounts.z() ? tails.z() : regularSegmentSize.z() + 1 );
351 
352  HUComponent ::initializeSegment( myGrid->segmentAt( segmentCoord ), segmentSize );
353  NormalsComponent::initializeSegment( myGrid->segmentAt( segmentCoord ), segmentSize );
354  }
355 }

401 template< typename SegmentHUVolumeType, typename SegmentNormalsVolumeType >
402 base::Node* VolumeGridHelper< SegmentHUVolumeType, SegmentNormalsVolumeType >::createNode
403  ( unsigned int geometryType, const Spacing& spacing, const Dimensions& dimensions ) const
404 {
405  /* Compute dimensions of a regular grid segment,
406  * taking the redundant texels into account.
407  */
408  const base::math::Vector3f regularSegmentDimensions = spacing.millimeters.cwiseProduct( regularSegmentSize.cast< float >() );
409 
410  /* Create pivot node that shifts it's children to a corner.
411  */
412  base::Node* const pivot = new base::Node();
413  pivot->localTransform = base::math::translation4f( ( regularSegmentDimensions - dimensions.millimeters ) / 2 );
414  pivot->setMovable( false );
415 
416  /* Create geometry nodes for all grid segments.
417  */
418  CARNA_FOR_VECTOR3UI( segmentCoord, myGrid->segmentCounts )
419  {
420  const base::VolumeSegment< SegmentHUVolumeType, SegmentNormalsVolumeType >& segment = myGrid->segmentAt( segmentCoord );
421 
422  /* Compute dimensions of particular grid segment.
423  */
424  const bool isTail =
425  segmentCoord.x() + 1 == myGrid->segmentCounts.x()
426  || segmentCoord.y() + 1 == myGrid->segmentCounts.y()
427  || segmentCoord.z() + 1 == myGrid->segmentCounts.z();
428  const base::math::Vector3ui& volumeSize = segment.huVolume().size;
429  const base::math::Vector3f dimensions = !isTail ? regularSegmentDimensions
430  : ( ( volumeSize.cast< int >() - base::math::Vector3i( 1, 1, 1 ) )
431  .cast< float >().cwiseProduct( spacing.millimeters ) );
432 
433  /* Create geometry node for particular grid segment.
434  */
435  base::Geometry* const geom = new base::Geometry( geometryType );
436  pivot->attachChild( geom );
437  HUComponent ::attachTexture( *geom, segment );
438  NormalsComponent::attachTexture( *geom, segment );
439  geom->setMovable( false );
440  geom->setBoundingVolume( new base::BoundingBox( 1, 1, 1 ) );
441  geom->localTransform
442  = base::math::translation4f( segmentCoord.cast< float >().cwiseProduct( regularSegmentDimensions )
443  - ( !isTail
444  ? base::math::Vector3i( 0, 0, 0 )
445  : regularSegmentDimensions.cast< int >() - dimensions.cast< int >() ).cast< float >() / 2 )
446  * base::math::scaling4f( dimensions );
447  }
448 
449  /* We're done.
450  */
451  return pivot;
452 }
